390CMR_GRAPH_ITER CMRgraphEdgesNext(
391 CMR_GRAPH* graph,
392 CMR_GRAPH_ITER i
393)
394{
395 CMR_GRAPH_NODE source = graph->arcs[i ^ 1].target;
396 CMR_GRAPH_ITER j = graph->arcs[i].next;
